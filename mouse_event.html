mouseover/mouseout 
	const container = document.querySelector('#container');

	// Создадим обработчик

	const handler = (event) => console.log(event.type, event.target, event.currentTarget);

	// Установим обработчик на mouseover / mouseout

	container.addEventListener('mouseover', handler);
	container.addEventListener('mouseout', handler);
mouseenter/mouseleave 
	container.removeEventListener('mouseover', handler);
	container.removeEventListener('mouseout', handler);
	
	container.addEventListener('mouseenter', handler);
	container.addEventListener('mouseleave', handler);
mousedown/mouseup 
	container.removeEventListener('mouseenter', handler);
	container.removeEventListener('mouseleave', handler);

	container.addEventListener('mousedown', handler);
	container.addEventListener('mouseup', handler);
mousemove 
	container.addEventListener('mousemove', handler);
