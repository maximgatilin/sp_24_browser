custom event 
	// Создаем кастомное событие

	const myCustomEvent = new CustomEvent('my-custom-event', {
		detail: {
			foo: 'bar'
		}
	});

	// Добавим обработчик на document

	document.addEventListener('my-custom-event', () => {
		console.log(event.type, 'fired on document!', event.detail);
	});

	// Вызовем событие на document

	console.log(document.dispatchEvent(myCustomEvent));
prevent default 
	// Создаем кастомное событие

	const myCustomEvent = new CustomEvent('my-custom-event', {
		cancelable: true,
		detail: {
			foo: 'bar'
		}
	});

	// Добавим обработчик на document

	document.addEventListener('my-custom-event', (event) => {
		console.log(event.type, 'fired on document!', event.detail);
		event.preventDefault();
	});

	// Вызовем событие на document

	console.log(document.dispatchEvent(myCustomEvent));
