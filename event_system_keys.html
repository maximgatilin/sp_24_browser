	const textarea = document.querySelector('#textarea');
	const firstHandler = (event) => {
		if (event.altKey || event.ctrlKey || event.shiftKey || event.metaKey) {
			console.log(
				'System keys:',
				event.altKey ? 'Alt' : '',
				event.ctrlKey ? 'Ctrl' : '',
				event.shiftKey ? 'Shift' : '',
				event.metaKey ? 'Meta' : '',
			);
		}
	}
	textarea.addEventListener('keydown', firstHandler);
sending message 
  	textarea.removeEventListener('keydown', firstHandler);
	const messageHander = (event) => {
		if (event.key === 'Enter' && !event.shiftKey) {
			event.preventDefault();
			console.log('Sending message: ', event.currentTarget.value);
			event.currentTarget.value = '';
		}
	}
	textarea.addEventListener('keydown', messageHander);
