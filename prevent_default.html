	const link = document.querySelector('#hh-link');
	link.addEventListener('click', (event) => {
		event.preventDefault();
		console.log('Link default behaviour is canceled');
	});
input 
	const input = document.querySelector('#input-text');
	input.addEventListener('keypress', (event) => {
		if (event.keyCode < 48 || event.keyCode > 57) {
			event.preventDefault();
			console.log('Only numbers are allowed for input');
		}
	});
checkbox 
	const checkInput = document.querySelector('#input-checkbox');
	checkInput.addEventListener('click', (event) => {
		if (event.altKey) {
			return;
		}
		event.preventDefault();
		console.log('Checkbox is working only with alt pressed');
	})
